def analyze_currency_changes(tracking_data, alert_threshold=0.5):
    """–ê–Ω–∞–ª–∏–∑ –∏–∑–º–µ–Ω–µ–Ω–∏–π –∏ —Å–æ–∑–¥–∞–Ω–∏–µ –∞–ª–µ—Ä—Ç–æ–≤"""
    
    analysis_results = {}
    
    for currency, data in tracking_data.items():
        if len(data['rates']) < 2:
            continue
            
        start_rate = data['rates'][0]
        end_rate = data['rates'][-1]
        max_rate = max(data['rates'])
        min_rate = min(data['rates'])
        
        change_pct = ((end_rate - start_rate) / start_rate) * 100
        volatility = ((max_rate - min_rate) / start_rate) * 100
        
        analysis_results[currency] = {
            'start_rate': start_rate,
            'end_rate': end_rate,
            'change_percent': change_pct,
            'volatility': volatility,
            'max_rate': max_rate,
            'min_rate': min_rate
        }
        
        # –°–æ–∑–¥–∞–µ–º –∞–ª–µ—Ä—Ç –ø—Ä–∏ –∑–Ω–∞—á–∏—Ç–µ–ª—å–Ω–æ–º –∏–∑–º–µ–Ω–µ–Ω–∏–∏
        if abs(change_pct) > alert_threshold:
            direction = "üìà –≤—ã—Ä–æ—Å" if change_pct > 0 else "üìâ —É–ø–∞–ª"
            print(f"üö® –ê–õ–ï–†–¢: {currency} {direction} –Ω–∞ {abs(change_pct):.2f}%!")
    
    return analysis_results

# –ê–Ω–∞–ª–∏–∑ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤

    print(f"   –ò–∑–º–µ–Ω–µ–Ω–∏–µ: {stats['change_percent']:+.2f}%")
