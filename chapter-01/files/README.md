## cafe-analysis.py
```python
# cafe-analysis.py
# Скрипт для анализа продаж кофе в кафе "Утро"

import pandas as pd  # импортируем pandas для работы с таблицами

# Загружаем данные из CSV-файла с кодировкой UTF-8 и разделителем ';'
# В файле используется запятая в качестве десятичного разделителя
file_path = 'cafe-sales.csv'
df = pd.read_csv(file_path, encoding='utf-8', sep=';', decimal=',')

# Рассчитываем общую выручку
# sum() суммирует значения в столбце 'Выручка,руб'
total_revenue = df['Выручка,руб'].sum()

# Рассчитываем среднюю цену за чашку
# Для этого делим выручку на число чашек и находим среднее значение
df['Цена за чашку'] = df['Выручка,руб'] / df['Чашек']
average_price = df['Цена за чашку'].mean()

# Определяем лучший и худший день по количеству чашек
best_day = df.loc[df['Чашек'].idxmax(), 'День недели']
worst_day = df.loc[df['Чашек'].idxmin(), 'День недели']

# Выводим результаты
print(f"Общая выручка: {total_revenue:.2f} руб")
print(f"Средняя цена за чашку: {average_price:.2f} руб")
print(f"Лучший день по продажам чашек: {best_day}")
print(f"Худший день по продажам чашек: {worst_day}")

# Строим текстовый график продаж
# Каждый '*' соответствует 10 проданным чашкам
print("\nТекстовый график продаж (каждый '*' = 10 чашек):")
for _, row in df.iterrows():
    stars = '*' * int(row['Чашек'] / 10)
    print(f"{row['День недели']}: {stars}")
```
