## 🐍 Готовые функции Python:

Если python-stats.py не доступен, используйте эти функции:

```python
def analyze_data(data, name="Показатель", unit=""):
    """Комплексный анализ любого набора данных"""
    import pandas as pd
    import numpy as np
    
    stats = {
        'Среднее': data.mean(),
        'Медиана': data.median(), 
        'Стд.откл': data.std(),
        'Минимум': data.min(),
        'Максимум': data.max(),
        'Размах': data.max() - data.min(),
        'Q1': data.quantile(0.25),
        'Q3': data.quantile(0.75),
        'IQR': data.quantile(0.75) - data.quantile(0.25)
    }
    
    print(f"\n📊 Анализ: {name}")
    print("="*40)
    
    for key, value in stats.items():
        if key in ['Q1', 'Q3', 'IQR']:
            continue
        print(f"{key}: {value:.1f} {unit}")
    
    # Поиск выбросов
    lower_bound = stats['Q1'] - 1.5 * stats['IQR'] 
    upper_bound = stats['Q3'] + 1.5 * stats['IQR']
    
    outliers = data[(data < lower_bound) | (data > upper_bound)]
    if len(outliers) > 0:
        print(f"\n⚠️ Найдено выбросов: {len(outliers)}")
        print(f"Значения: {list(outliers.round(1))}")
    
    return stats

def compare_groups(group1, group2, name1="Группа 1", name2="Группа 2"):
    """Сравнение двух групп данных"""
    
    print(f"\n🔍 Сравнение: {name1} vs {name2}")
    print("="*50)
    
    stats1 = analyze_data(group1, name1)
    stats2 = analyze_data(group2, name2)
    
    # Сравнение средних
    diff_mean = ((stats2['Среднее'] - stats1['Среднее']) / stats1['Среднее'] * 100)
    print(f"\n📈 Разница в средних: {diff_mean:+.1f}%")
    
    # Сравнение стабильности  
    cv1 = stats1['Стд.откл'] / stats1['Среднее'] * 100
    cv2 = stats2['Стд.откл'] / stats2['Среднее'] * 100
    
    print(f"📊 Коэффициент вариации:")
    print(f"  {name1}: {cv1:.1f}%")
    print(f"  {name2}: {cv2:.1f}%")
    
    if cv1 < cv2:
        print(f"  → {name1} стабильнее")
    else:
        print(f"  → {name2} стабильнее")
```

## ❓ Часто задаваемые вопросы:

### Excel показывает ошибки в формулах?
- Проверьте локализацию: используйте СРЗНАЧ вместо AVERAGE
- Убедитесь что данные в числовом формате
- Проверьте разделители: точка или запятая в десятичных дробях

### Python выдаёт ошибки при загрузке CSV?
```python
# Попробуйте разные варианты кодировки
df = pd.read_csv('file.csv', encoding='utf-8')
# или
df = pd.read_csv('file.csv', encoding='cp1251') 
# или
df = pd.read_csv('file.csv', encoding='utf-8-sig')
```

### Данные выглядят странно?
- Проверьте разделители (запятая или точка с запятой)
- Убедитесь что первая строка содержит заголовки
- Проверьте что числа не содержат пробелы или лишние символы
