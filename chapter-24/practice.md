def create_biplot(components, feature_names, pca_model):
    plt.figure(figsize=(12, 8))
    
    # –¢–æ—á–∫–∏ (–∫–ª–∏–µ–Ω—Ç—ã) –≤ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç

    # –í–µ–∫—Ç–æ—Ä—ã –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö
    for i, (feature, loading) in enumerate(zip(feature_names, pca_model.components_.T)):
        plt.arrow(0, 0, loading[0]*3, loading[1]*3, 
                 head_width=0.05, head_length=0.05, 
                 fc='red', ec='red', alpha=0.8)

                fontsize=10, ha='center', va='center')
    
    plt.xlabel(f'PC1 ({pca.explained_variance_ratio_[0]:.1%} –∏–∑–º–µ–Ω—á–∏–≤–æ—Å—Ç–∏)')
    plt.ylabel(f'PC2 ({pca.explained_variance_ratio_[1]:.1%} –∏–∑–º–µ–Ω—á–∏–≤–æ—Å—Ç–∏)')
    plt.title('üìä Biplot: –ö–ª–∏–µ–Ω—Ç—ã –∏ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ –≤ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç')
